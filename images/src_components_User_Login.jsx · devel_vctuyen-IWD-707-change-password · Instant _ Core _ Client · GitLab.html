/**
 * Login
 */
/*global grecaptcha*/
import React, { Component } from 'react';
import Captcha from '../Captcha';
import { Form, Input, Button, Alert } from 'antd';
import CustomIcon from '../CustomIcon'
import { checkConnect } from '../../utils/connectInternet';

import './styles.less';

const FormItem = Form.Item;

class AuthLogin extends Component {

    constructor(props) {
        super(props);

        this.state = {
            captcha: '',
            captchaWidget: null,
            showAlert: false,
            alertMessage: '',
            isOnline: true,
            showHidePassword: false,
        }
    }

    handleForgotPassword = () => {
        const { onForgot } = this.props;
        onForgot();
    }

    handleSubmit = (e) => {
        e.preventDefault();
        const { onLogin, form } = this.props;
        const { captcha, captchaWidget, isOnline } = this.state;
        this.setState({
            isOnline: checkConnect(),
        })
        if (isOnline) {
            if (!captcha) {
                this.setState({
                    showAlert: true,
                    alertMessage: "Bạn chưa xác nhận reCaptcha",
                })
            } else {
                form.validateFields((err, values) => {
                    if (!err) {
                        onLogin(values.email, values.password, captcha);
                        if (grecaptcha && captchaWidget !== null) {
                            grecaptcha.reset(this.captchaWidget)
                            this.setState({
                                showAlert: false,
                                alertMessage: "",
                            })
                        }
                    }
                });
            }
        }
    }

    onchangeCaptcha = (widgetId, response) => {
        this.setState({
            captcha: response,
            captchaWidget: widgetId,
        })
    }
    changeIcPassword = () => {
        const { showHidePassword } = this.state;
        this.setState({
            showHidePassword: !showHidePassword
        })
    }

    render() {
        const { form, loading, alertShow, errMessage } = this.props;

        const { showAlert, alertMessage, isOnline, showHidePassword } = this.state;
        console.log(showHidePassword);
        const { getFieldDecorator } = form;
        return (
            <Form hideRequiredMark={true} onSubmit={this.handleSubmit} className="main-login">
                <div className="wrapContent">
                    <div className="title">
                        <p className="login-title">ĐĂNG NHẬP</p>
                        <div className="alert">
                            {(showAlert || alertShow) && isOnline ? <Alert
                                className="alert-message"
                                message={showAlert ? alertMessage : errMessage}
                                type="danger"
                            /> : <span />}
                        </div>
                        {
                            isOnline ? null : <div className="alert">
                                <Alert
                                    className="alert-message"
                                    message={`Vui lòng kiểm tra lại kết nối.`}
                                    type="danger"
                                />
                            </div>
                        }
                    </div>
                    <FormItem className="username-input">
                        <label>Tài khoản</label>
                        {getFieldDecorator('email', {
                            rules: [{ required: true, message: 'Tài khoản không được để trống' }],
                        })(
                            <Input
                                className="input-login"
                                placeholder="Tài khoản"
                                prefix={<CustomIcon type="ic-email" primaryColor="#707a86" size={20} />}
                                type="text"
                                disabled={loading}
                                autoFocus
                            />)}
                    </FormItem>

                    <FormItem className="password-input">
                        <label>Mật khẩu</label>
                        {getFieldDecorator('password', {
                            rules: [{ required: true, message: 'Mật khẩu không được để trống' }, { min: 8, message: 'Mật khẩu phải có độ dài từ 8 kí tự trở lên' }],
                        })(<Input
                            suffix={ <div className="wrapIcPassword" onClick={this.changeIcPassword}>
                            {showHidePassword ? <CustomIcon type="ic-show-pw" primaryColor="#707a86" size={20} /> : <CustomIcon type="ic-hide-pw" primaryColor="#707a86" size={20} />}
                        </div>}
                            type={showHidePassword ? "text" : "password"}
                            className="input-login"
                            placeholder="Mật khẩu"
                            prefix={<CustomIcon type="ic-password" primaryColor="#707a86" size={20} />}
                            disabled={loading}
                        />)}
                       
                    </FormItem>
                    <Captcha className="reCaptcha" onChange={this.onchangeCaptcha} />
                    <Button loading={loading} type="primary" htmlType="submit" className="btn">
                        Đăng nhập
                </Button>
                </div>
                <div // eslint-disable-line
                    className="forgot-password"
                >
                    Bạn đã quên mật khẩu? <label onClick={() => this.handleForgotPassword()} className="ktl" > Khởi tạo lại</label>
                </div>
            </Form>
        );
    }
}

export default Form.create()(AuthLogin);
